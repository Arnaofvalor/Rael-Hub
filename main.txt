local function criarInterface(texto, textbuttom, MapId)
    -- Cria a tela preta
    local screenGui = Instance.new("ScreenGui")
    screenGui.Name = "RaelHubGui"
    screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

    -- Cria o quadro principal
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0.8, 0, 0.8, 0)
    frame.Position = UDim2.new(0.1, 0, 0.1, 0)
    frame.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
    frame.BorderSizePixel = 0
    frame.ClipsDescendants = true
    frame.Parent = screenGui

    -- Adiciona borda arredondada ao frame
    local cornerFrame = Instance.new("UICorner")
    cornerFrame.CornerRadius = UDim.new(0, 12)
    cornerFrame.Parent = frame

    -- Adiciona padding ao frame
    local padding = Instance.new("UIPadding")
    padding.PaddingTop = UDim.new(0, 20)
    padding.PaddingBottom = UDim.new(0, 20)
    padding.PaddingLeft = UDim.new(0, 20)
    padding.PaddingRight = UDim.new(0, 20)
    padding.Parent = frame

    -- Cria o título
    local title = Instance.new("TextLabel")
    title.Size = UDim2.new(1, 0, 0.2, 0)
    title.Position = UDim2.new(0, 0, 0, 0)
    title.BackgroundTransparency = 1
    title.TextColor3 = Color3.fromRGB(34, 168, 110)
    title.TextSize = 36
    title.Text = "Rael Hub"
    title.TextStrokeTransparency = 0.8
    title.TextScaled = true
    title.Parent = frame

    -- Cria o parágrafo
    local paragraph = Instance.new("TextLabel")
    paragraph.Size = UDim2.new(1, 0, 0.4, 0)
    paragraph.Position = UDim2.new(0, 0, 0.2, 0)
    paragraph.BackgroundTransparency = 1
    paragraph.TextColor3 = Color3.fromRGB(255, 255, 255)
    paragraph.TextSize = 20
    paragraph.Text = texto
    paragraph.TextStrokeTransparency = 0.8
    paragraph.TextWrapped = true
    paragraph.Parent = frame

    -- Cria o botão
    local button = Instance.new("TextButton")
    button.Size = UDim2.new(0.5, 0, 0.2, 0)
    button.Position = UDim2.new(0.25, 0, 0.6, 0)
    button.BackgroundColor3 = Color3.fromRGB(34, 168, 110)
    button.TextColor3 = Color3.fromRGB(255, 255, 255)
    button.TextSize = 24
    button.Text = textbuttom
    button.TextStrokeTransparency = 0.8
    button.BorderSizePixel = 0
    button.BackgroundTransparency = 0
    button.BorderColor3 = Color3.fromRGB(255, 255, 255)
    button.Parent = frame

    -- Adiciona borda arredondada ao botão
    local cornerButton = Instance.new("UICorner")
    cornerButton.CornerRadius = UDim.new(0, 12)
    cornerButton.Parent = button

    -- Cria o botão de fechar (X)
    local closeButton = Instance.new("TextButton")
    closeButton.Size = UDim2.new(0.1, 0, 0.1, 0)
    closeButton.Position = UDim2.new(0.9, 0, 0, 0)
    closeButton.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    closeButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    closeButton.TextSize = 24
    closeButton.Text = "X"
    closeButton.TextStrokeTransparency = 0.8
    closeButton.BorderSizePixel = 0
    closeButton.BackgroundTransparency = 0
    closeButton.BorderColor3 = Color3.fromRGB(255, 255, 255)
    closeButton.ZIndex = 2
    closeButton.Parent = frame

    -- Adiciona borda arredondada ao botão de fechar
    local cornerCloseButton = Instance.new("UICorner")
    cornerCloseButton.CornerRadius = UDim.new(0, 6)
    cornerCloseButton.Parent = closeButton

    -- Adiciona efeito de hover
    closeButton.MouseEnter:Connect(function()
        closeButton.BackgroundColor3 = Color3.fromRGB(200, 0, 0)
    end)
    closeButton.MouseLeave:Connect(function()
        closeButton.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    end)

    -- Função para fechar a interface
    local function fecharInterface()
        screenGui:Destroy()
    end

    -- Conecta a função ao clique do botão de fechar
    closeButton.MouseButton1Click:Connect(fecharInterface)

    -- Função para teleportar o jogador
    local function entrarNoJogo()
        local placeId = MapId -- ID do mapa que você quer entrar
        game:GetService("TeleportService"):Teleport(placeId, game.Players.LocalPlayer)
    end

    -- Conecta a função ao clique do botão
    button.MouseButton1Click:Connect(entrarNoJogo)

    -- Adiciona o TweenService
    local tweenService = game:GetService("TweenService")

    -- Função para mudar a cor do título
    local function mudarCorDoTitulo()
        local cores = {
            Color3.fromRGB(47, 236, 154),  -- Cor1: #2FEC9A
            Color3.fromRGB(34, 168, 110),  -- Cor2: #22A86E
            Color3.fromRGB(27, 134, 88),   -- Cor3: #1B8658
            Color3.fromRGB(8, 40, 26)      -- Cor4: #08281A
        }

        local tweenInfo = TweenInfo.new(2, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut, -1, true)  -- Tempo aumentado para suavizar a transição
        local tweenGoal = {TextColor3 = cores[1]}
        
        local tween = tweenService:Create(title, tweenInfo, tweenGoal)
        tween:Play()
        
        -- Alterna as cores
        while true do
            for _, cor in ipairs(cores) do
                tweenGoal.TextColor3 = cor
                tween:Play()
                wait(2)  -- Tempo de exibição da cor antes de mudar
            end
        end
    end

    -- Inicia a mudança de cor
    mudarCorDoTitulo()
end
