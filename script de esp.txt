


-- Carrega a biblioteca Orion
local OrionLib = loadstring(game:HttpGet('https://raw.githubusercontent.com/shlexware/Orion/main/source'))()

-- Carrega o módulo ESP
local espModule = loadstring(game:HttpGet('https://raw.githubusercontent.com/Laelmano24/Script-de-the-mimic-book-2-chapter-3-basico/main/esp.txt'))()

local Window = OrionLib:MakeWindow({
    Name = "Rael Hub",
    HidePremium = false,
    SaveConfig = true,
    ConfigFolder = "OrionTest"
})

local Aba1 = Window:MakeTab({
    Name = "Esp hiachi",
    Icon = "rbxassetid://4483345998",
    PremiumOnly = false
})

local Aba1_section1 = Aba1:AddSection({
    Name = "Esp"
})

-- Variável para controlar o estado do ESP do monstro Hiachi
local espHiachiEnabled = false

-- Função para alternar o estado do ESP do monstro Hiachi
local function toggleESPHiachi()
    espHiachiEnabled = not espHiachiEnabled
    if espHiachiEnabled then
        espModule.activateESPHiachi()
        OrionLib:MakeNotification({
            Name = "ESP Hiachi Ativado",
            Content = "O ESP foi ativado para o monstro Hiachi.",
            Image = "rbxassetid://4483345998",
            Time = 5
        })
    else
        espModule.deactivateESPHiachi()
        OrionLib:MakeNotification({
            Name = "ESP Hiachi Desativado",
            Content = "O ESP foi desativado para o monstro Hiachi.",
            Image = "rbxassetid://4483345998",
            Time = 5
        })
    end
end

Aba1_section1:AddButton({
    Name = "Alternar ESP Hiachi",
    Callback = function()
        toggleESPHiachi()
    end    
})

-- Variável para controlar o estado do ESP dos jogadores
local espPlayersEnabled = false

-- Função para alternar o estado do ESP dos jogadores
local function toggleESPPlayers()
    espPlayersEnabled = not espPlayersEnabled
    if espPlayersEnabled then
        espModule.activateESPPlayers()
        OrionLib:MakeNotification({
            Name = "ESP Jogadores Ativado",
            Content = "O ESP foi ativado para todos os jogadores.",
            Image = "rbxassetid://4483345998",
            Time = 5
        })
    else
        espModule.deactivateESPPlayers()
        OrionLib:MakeNotification({
            Name = "ESP Jogadores Desativado",
            Content = "O ESP foi desativado para todos os jogadores.",
            Image = "rbxassetid://4483345998",
            Time = 5
        })
    end
end

Aba1_section1:AddButton({
    Name = "Alternar ESP Jogadores",
    Callback = function()
        toggleESPPlayers()
    end    
})

OrionLib:Init()
