local function criarInterface(texto, textbuttom, MapId, MusicId)
    
    wait(1)
    
    local screenGui = Instance.new("ScreenGui")
    screenGui.Name = "RaelHubGui"
    screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

    -- Adiciona o som
    local sound = Instance.new("Sound")
    sound.SoundId = MusicId
    
    sound.Volume = 1
    sound.Parent = screenGui
    sound:Play()

    -- Cria o quadro principal
    local frame = Instance.new("Frame")
    frame.Size = UDim2.new(0.6, 0, 0.6, 0)
    frame.Position = UDim2.new(0.2, 0, 0.2, 0)
    frame.BackgroundTransparency = 1
    frame.BorderSizePixel = 0
    frame.ClipsDescendants = true
    frame.Parent = screenGui

    -- Adiciona a imagem de fundo
    local backgroundImage = Instance.new("ImageLabel")
    backgroundImage.Size = UDim2.new(1, 0, 1, 0)
    backgroundImage.Position = UDim2.new(0, 0, 0, 0)
    backgroundImage.BackgroundTransparency = 1
    backgroundImage.Image = "rbxassetid://18665679839"
    backgroundImage.ScaleType = Enum.ScaleType.Crop
    backgroundImage.Parent = frame

    -- Adiciona borda arredondada ao frame
    local cornerFrame = Instance.new("UICorner")
    cornerFrame.CornerRadius = UDim.new(0, 12)
    cornerFrame.Parent = frame

    -- Cria o título
    local title = Instance.new("TextLabel")
    title.Size = UDim2.new(1, 0, 0.15, 0)
    title.Position = UDim2.new(0, 0, 0.05, 0)
    title.BackgroundTransparency = 1
    title.TextColor3 = Color3.fromRGB(34, 168, 110)
    title.TextSize = 36
    title.Text = "Rael Hub"
    title.TextStrokeTransparency = 0.8
    title.TextScaled = true
    title.Parent = frame

    -- Cria o parágrafo
    local paragraph = Instance.new("TextLabel")
    paragraph.Size = UDim2.new(1, 0, 0.3, 0)
    paragraph.Position = UDim2.new(0, 0, 0.2, 0)
    paragraph.BackgroundTransparency = 1
    paragraph.TextColor3 = Color3.fromRGB(255, 255, 255)
    paragraph.TextSize = 12
    paragraph.Text = texto
    paragraph.TextStrokeTransparency = 0.8
    paragraph.TextWrapped = true
    paragraph.Parent = frame

    -- Cria o botão
    local button = Instance.new("TextButton")
    button.Size = UDim2.new(0.5, 0, 0.15, 0)
    button.Position = UDim2.new(0.25, 0, 0.55, 0)
    button.BackgroundColor3 = Color3.fromRGB(34, 168, 110)
    button.TextColor3 = Color3.fromRGB(255, 255, 255)
    button.TextSize = 16
    button.Text = textbuttom
    button.TextStrokeTransparency = 0.8
    button.BorderSizePixel = 0
    button.Parent = frame

    -- Adiciona borda arredondada ao botão
    local cornerButton = Instance.new("UICorner")
    cornerButton.CornerRadius = UDim.new(0, 12)
    cornerButton.Parent = button

    -- Cria o botão de fechar (X)
    local closeButton = Instance.new("TextButton")
    closeButton.Size = UDim2.new(0.1, 0, 0.1, 0)
    closeButton.Position = UDim2.new(0.9, -30, 0.05, 0)
    closeButton.BackgroundTransparency = 1 -- Fundo invisível
    closeButton.TextColor3 = Color3.fromRGB(168, 34, 34) -- Texto vermelho escuro (#A82222)
    closeButton.TextSize = 18 -- Tamanho do texto reduzido
    closeButton.Text = "X"
    closeButton.TextStrokeTransparency = 0.8
    closeButton.BorderSizePixel = 0
    closeButton.ZIndex = 2
    closeButton.Parent = frame

    -- Adiciona borda arredondada ao botão de fechar
    local cornerCloseButton = Instance.new("UICorner")
    cornerCloseButton.CornerRadius = UDim.new(0, 6)
    cornerCloseButton.Parent = closeButton

    -- Função para fechar a interface
    local function fecharInterface()
        screenGui:Destroy()
    end

    -- Conecta a função ao clique do botão de fechar
    closeButton.MouseButton1Click:Connect(fecharInterface)

    -- Função para teleportar o jogador
    local function entrarNoJogo()
        game:GetService("TeleportService"):Teleport(MapId, game.Players.LocalPlayer)
    end

    -- Conecta a função ao clique do botão
    button.MouseButton1Click:Connect(entrarNoJogo)

    -- Adiciona o TweenService para mudar a cor do título
    local tweenService = game:GetService("TweenService")

    local function mudarCorDoTitulo()
        local cores = {
            Color3.fromRGB(47, 236, 154),
            Color3.fromRGB(34, 168, 110),
            Color3.fromRGB(27, 134, 88),
            Color3.fromRGB(8, 40, 26)
        }

        local tweenInfo = TweenInfo.new(2, Enum.EasingStyle.Linear, Enum.EasingDirection.InOut, -1, true)
        local tweenGoal = {TextColor3 = cores[1]}
        
        local tween = tweenService:Create(title, tweenInfo, tweenGoal)
        tween:Play()
        
        -- Alterna as cores
        while true do
            for _, cor in ipairs(cores) do
                tweenGoal.TextColor3 = cor
                tween:Play()
                wait(2)
            end
        end
    end

    -- Inicia a mudança de cor
    mudarCorDoTitulo()
end



local function showImageWithSound()
   wait(1)
    -- IDs fixos
    local imageId = ""
    
    
    local Players = game:GetService("Players")
    local player = Players.LocalPlayer

    local locale = player.LocaleId:lower()

    if locale == "pt-br" then
      
      imageId = "rbxassetid://18655033752"
    
    elseif locale == "en-us" then
      
      imageId = "rbxassetid://18654153605"
    
    elseif locale:sub(1, 2) == "es" then
    
      imageId = "rbxassetid://18655089320"
    else
      
      imageId = "rbxassetid://18654153605"
    
    end
    
    
    local soundId = "rbxassetid://6073491164"
    local fadeDuration = 0.2
    local waitTime = 3

    -- Cria a ScreenGui e a adiciona ao CoreGui
    local screenGui = Instance.new("ScreenGui")
    screenGui.Parent = game.CoreGui

    -- Cria o ImageLabel e define suas propriedades
    local imageLabel = Instance.new("ImageLabel")
    imageLabel.Parent = screenGui

    -- Define o tamanho da imagem como 50% da tela
    imageLabel.Size = UDim2.new(0.5, 0, 0.5, 0) -- 50% da largura e 50% da altura da tela

    -- Define o ponto de ancoragem no centro da imagem
    imageLabel.AnchorPoint = Vector2.new(0.5, 0.5)

    -- Centraliza a imagem na tela
    imageLabel.Position = UDim2.new(0.5, 0, 0.5, 0) -- Centraliza no meio da tela, ajustado pelo AnchorPoint

    -- Define o ID da imagem e remove o fundo
    imageLabel.Image = imageId
    imageLabel.BackgroundTransparency = 1

    -- Ajusta a imagem para manter suas proporções e não esticar
    imageLabel.ScaleType = Enum.ScaleType.Fit

    -- Cria e configura o som
    local sound = Instance.new("Sound")
    sound.SoundId = soundId
    sound.Parent = game.SoundService -- Adiciona o som ao serviço de som
    sound.Volume = 1 -- Define o volume (0 a 1)
    sound.Looped = false -- Não repete o som
    sound:Play() -- Reproduz o som

    -- Função para fazer o fade in
    local function fadeIn(duration)
        local increment = 0.05
        local step = increment / duration
        for i = 0, 1, step do
            imageLabel.ImageTransparency = 1 - i
            wait(increment)
        end
    end

    -- Função para fazer o fade out
    local function fadeOut(duration)
        local increment = 0.05
        local step = increment / duration
        for i = 0, 1, step do
            imageLabel.ImageTransparency = i
            wait(increment)
        end
    end

    -- Faz o fade in da imagem
    fadeIn(fadeDuration)

    -- Aguarda o tempo especificado após o fade in
    wait(waitTime)

    -- Faz o fade out da imagem
    fadeOut(fadeDuration)

    -- Remove o ImageLabel e o som após o fade out
    imageLabel:Destroy()
    sound:Destroy()
    
end
