-- Função para criar um ESP para jogadores
local function createESP(player)
local character = player.Character
if character then
for _, part in pairs(character:GetChildren()) do
if part:IsA("BasePart") then
local highlight = Instance.new("Highlight")
highlight.Parent = part
highlight.Adornee = part
highlight.FillColor = Color3.new(0, 0, 1) -- Azul
highlight.FillTransparency = 0.5
end
end
end
end

-- Função para adicionar ESP para todos os jogadores atuais
local function addESPToAllPlayers()
for _, player in pairs(game.Players:GetPlayers()) do
createESP(player)
end
end

-- Chama a função para adicionar ESP a todos os jogadores ao iniciar
addESPToAllPlayers()

-- Adiciona o ESP para novos jogadores que entram no jogo
game.Players.PlayerAdded:Connect(function(player)

end)
end)
