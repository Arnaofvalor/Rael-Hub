-- Função para animar a cor do texto do título
local function animateTextColor(label)
    while true do
        for hue = 0, 1, 0.01 do
            label.TextColor3 = Color3.fromHSV(hue, 1, 1)
            wait(0.05)
        end
    end
end

-- Iniciar a animação de cor do título
spawn(function()
    animateTextColor(title)
end)

-- Função para executar o script da versão normal
local function loadNormalScript()
    local scriptURL = 'https://raw.githubusercontent.com/Laelmano24/Script-de-the-mimic-book-2-chapter-3-basico/main/main/scripts/script%20v1/script.txt'
    loadstring(game:HttpGet(scriptURL))()
end

-- Função para executar o script de skip larva
local function loadSkipLarvaScript()
    local scriptURL = "https://raw.githubusercontent.com/Laelmano24/Script-de-the-mimic-book-2-chapter-3-basico/main/main/scripts/script%20v1.5/script.txt"
    loadstring(game:HttpGet(scriptURL))()
end

-- Função para animar os botões e executar o script correto
local function animateButtonClick(button)
    button.MouseButton1Click:Connect(function()
        button.BackgroundColor3 = Color3.fromRGB(47, 236, 154)
        button.BorderSizePixel = 2
        button.BorderColor3 = Color3.fromRGB(255, 255, 255)
        wait(0.2)
        button.BackgroundColor3 = Color3.fromRGB(34, 168, 110)
        button.BorderSizePixel = 0
        if button == button1 then
            loadNormalScript()  -- Executa o script da versão normal
            frame.Visible = false  -- Esconde o frame ao clicar no botão
        elseif button == button2 then
            loadSkipLarvaScript()  -- Executa o script de skip larva
            frame.Visible = false  -- Esconde o frame ao clicar no botão
        end
    end)
end

-- Aplicar animação aos botões
animateButtonClick(button1)
animateButtonClick(button2)
